//Difficulty: Medium
//Problem Number: 11

class Solution {
public:
    int maxArea(vector<int>& height) {
        int max = 0;
        int i = 0, j = height.size() - 1;
        while(i < j) {
            int area = min({height[i], height[j]}) * (j-i);
            if(area > max) 
                max = area;
            if(height[i] > height[j]) 
                j--;
            else i++;
        }
        return max;      
    }
};
